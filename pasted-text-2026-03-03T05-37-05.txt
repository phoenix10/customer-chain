全自定义自动报价系统（纯代码开发，无低代码依赖）

一、系统核心设计（通用化核心）

核心目标：让用户（管理员）无需修改代码，即可自行配置「产品参数、计算公式、报价组成、BOM物料、开模规则」，业务员仅需选择配置好的模板，输入参数即可自动报价。整体架构仍采用前后端分离（Python+FastAPI+PostgreSQL+原生前端），重点新增「配置模块、BOM物料模块、开模规则模块」，实现全流程自定义，适配小批量开模、BOM物料灵活引用场景。

1. 通用化设计核心思路

- 参数自定义：用户可新增任意产品类型，为每种产品配置专属参数（如尺寸、基重、淋膜量、厚度等，支持数字/文本/下拉类型）；

- 公式自定义：用户可通过可视化/文本方式，为产品配置核心计算公式（如克重、单耗、物料成本公式），支持参数变量、基础运算、函数调用、BOM物料引用；

- 报价组成自定义：用户可自由配置报价的成本项（如物料成本、工艺成本、运费、税费、利润、开模费），每个成本项可绑定自定义计算规则，开模费支持按订单数量阈值触发；

- 模板化管理：将「产品参数+计算公式+报价组成+BOM物料+开模规则」封装为报价模板，业务员直接选择模板、输入参数，系统自动执行计算（含开模费判断、BOM物料引用），无需关注底层逻辑；

- BOM物料灵活引用：新增BOM物料管理，支持物料分类、物料单价维护，公式可直接引用BOM物料单价/用量，实现物料成本灵活计算；

- 开模费规则自定义：支持配置开模费阈值（如5000数量以下）、开模费金额/计算公式，系统自动根据订单数量判断是否触发开模费，无需人工干预。

2. 系统整体架构（新增配置层+BOM层+开模规则层）

新增「配置管理模块」「模板管理模块」「BOM物料管理模块」「开模规则管理模块」，作为系统通用化的核心；其中BOM物料模块负责物料维护与引用，开模规则模块负责小批量开模费触发逻辑，所有自定义规则均通过这四个模块存储和调用，核心业务层仅负责执行规则，无需硬编码任何产品/公式/物料/开模逻辑。

-- 2. 参数配置表（用户为每种产品配置专属参数，如长度、基重等）
CREATE TABLE param_config (
  param_id SERIAL PRIMARY KEY COMMENT '参数ID',
  type_id INT NOT NULL COMMENT '关联产品类型ID',
  param_name VARCHAR(50) NOT NULL COMMENT '参数名称（如长度、基重）',
  param_type VARCHAR(20) NOT NULL COMMENT '参数类型（number：数字/text：文本/select：下拉）',
  param_unit VARCHAR(20) COMMENT '参数单位（如cm、g/㎡）',
  param_options JSONB COMMENT '下拉参数的选项（JSON格式，如["单面","双面"]）',

  create_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  update_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (type_id) REFERENCES product_type(type_id) ON DELETE CASCADE

-- 3. 公式配置表（用户自定义计算公式，如克重、物料成本公式，支持引用BOM物料）
CREATE TABLE formula_config (

  formula_desc TEXT COMMENT '公式描述',
  param_ids JSONB NOT NULL COMMENT '关联的参数ID（JSON格式，如[1,2,3]）',
  bom_ids JSONB COMMENT '关联的BOM物料ID（JSON格式，如[1,2]，可选）',
  result_unit VARCHAR(20) COMMENT '计算结果单位（如g、元）',
  create_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  update_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 4. 工艺配置表（用户可新增任意工艺，自定义加价规则）
CREATE TABLE process_config (
  process_id SERIAL PRIMARY KEY COMMENT '工艺ID',
  process_name VARCHAR(50) NOT NULL COMMENT '工艺名称（如淋膜、模切）',
  add_type VARCHAR(20) NOT NULL COMMENT '加价方式（area/unit/ratio：按面积/按单张/按比例）',

# -------------------------- 1. 工具函数：公式解析（核心，支持自定义公式+BOM物料引用） --------------------------
def parse_formula(formula_expr: str, param_dict: Dict[str, Any], bom_dict: Dict[str, Any] = None) -> float:
    """
    解析自定义公式，代入参数值、BOM物料值计算结果
    :param formula_expr: 公式表达式（字符串，如"(length*width/10000)*base_weight"，支持bom_price_1等BOM引用）
    :param param_dict: 参数字典（key=参数名称，value=参数值，如{"length":10, "width":8}）
    :param bom_dict: BOM物料字典（key=bom_price_+bom_id，value=物料单价，如{"bom_price_1":8500, "bom_price_2":12000}）
    :return: 计算结果（保留4位小数）
    """
    try:
        # 替换公式中的参数为实际值（支持嵌套公式结果）
        expr = formula_expr
        # 先替换参数值
        for param_name, param_value in param_dict.items():
            # 确保参数值为数字（避免文本参数干扰）
            try:
                param_value = float(param_value)
                expr = expr.replace(param_name, str(param_value))
            except (ValueError, TypeError):
                continue
        # 再替换BOM物料单价（若有）
        if bom_dict:
            for bom_key, bom_price in bom_dict.items():
                try:
                    bom_price = float(bom_price)
                    expr = expr.replace(bom_key, str(bom_price))
                except (ValueError, TypeError):
                    continue
        # 解析公式并计算（使用sympy避免eval的安全风险）
        result = sympify(expr).evalf()
        return round(float(result), 4)
    except Exception as e:
        raise HTTPException(status_code=400, detail=f"公式解析失败：{str(e)}，公式表达式：{formula_expr}")

# -------------------------- 2. 工具函数：开模费计算（核心，适配5000数量以下加开模费） --------------------------
def calculate_mold_fee(order_quantity: int, mold_rule: Dict[str, Any], param_dict: Dict[str, Any]) -> float:
    """
    根据订单数量和开模规则，计算开模费
    :param order_quantity: 订单数量
    :param mold_rule: 开模规则字典（含阈值、开模费类型、金额/公式）
    :param param_dict: 参数字典（用于公式计算开模费时代入参数）
    :return: 开模费金额（保留2位小数）
    """
    if not mold_rule:
        return 0.0  # 无开模规则，不收取开模费
    min_quantity = mold_rule["min_quantity"]
    # 判断是否触发开模费（订单数量低于阈值）
    if order_quantity >= min_quantity:
        return 0.0
    # 计算开模费
    mold_fee_type = mold_rule["mold_fee_type"]
    if mold_fee_type == "fixed":
        # 固定金额开模费
        return round(float(mold_rule["mold_fee_value"]), 2)
    elif mold_fee_type == "formula":
        # 公式计算开模费
        mold_fee_expr = mold_rule["mold_fee_expr"]
        mold_fee = parse_formula(mold_fee_expr, param_dict)
        return round(mold_fee, 2)
    else:
        raise HTTPException(status_code=400, detail=f"不支持的开模费类型：{mold_fee_type}")

# -------------------------- 9. 核心模块5：报价计算（调用自定义配置+BOM+开模规则，自动执行） --------------------------
def calculate_quote_by_template(params: QuoteCalculateParams, db: psycopg2.extensions.connection) -> Dict[str, Any]:
    """
    根据用户选择的模板，输入的参数，自动执行报价计算（含BOM引用、开模费判断）
    :param params: 前端传入参数（包含template_id、param_values、process_ids、bom_ids等）
    :param db: 数据库连接
    :return: 报价结果（字典）
    """
    # 1. 解析参数
    template_id = params.template_id
    param_values = params.param_values  # 输入的参数值（key=param_name，value=值）
    process_ids = params.process_ids if params.process_ids else []  # 选中的工艺ID
    bom_ids = params.bom_ids if params.bom_ids else []  # 选中的BOM物料ID（灵活选择，优先级高于模板默认BOM）
    order_quantity = int(param_values.get("order_quantity", 1))  # 订单数量（默认1）

    # 2. 查询模板关联的所有配置
    with db.cursor(cursor_factory=psycopg2.extras.DictCursor) as cur:
        # 2.1 查询模板信息（含关联的BOM、开模规则）
        cur.execute("""
            SELECT t.*, mr.*
            FROM quote_template t
            LEFT JOIN mold_rule mr ON t.mold_rule_id = mr.rule_id
            WHERE t.template_id = %s
        """, (template_id,))
        template = cur.fetchone()
        if not template:
            raise HTTPException(status_code=404, detail="报价模板不存在")
        # 解析模板关联的ID列表（PostgreSQL JSONB直接转换为Python列表）
        formula_ids = template["formula_ids"]
        component_ids = template["component_ids"]
        type_id = template["type_id"]
        # 处理BOM物料ID：前端手动选择的BOM优先级高于模板默认BOM
        final_bom_ids = bom_ids if bom_ids else (template["bom_ids"] if template["bom_ids"] else [])
        # 处理开模规则：模板关联的开模规则
        mold_rule = dict(template) if template["rule_id"] else None  # 开模规则信息

        # 2.2 查询产品类型对应的参数配置（校验输入参数是否完整）
        cur.execute("SELECT * FROM param_config WHERE type_id = %s", (type_id,))
        param_configs = cur.fetchall()
        # 校验必填参数
        for param in param_configs:
            if param["is_required"] == 1 and param["param_name"] not in param_values:
                raise HTTPException(status_code=400, detail=f"必填参数缺失：{param['param_name']}")
            # 补充默认值（若未输入）
            if param["param_name"] not in param_values and param["default_value"]:
                param_values[param["param_name"]] = param["default_value"]

        # 2.3 查询模板关联的公式配置
        if formula_ids:
            placeholders = ", ".join(["%s"] * len(formula_ids))
            cur.execute(f"SELECT * FROM formula_config WHERE formula_id IN ({placeholders})", formula_ids)
        else:
            cur.execute("SELECT * FROM formula_config WHERE 1=0")
        formulas = cur.fetchall()

        # 2.4 查询模板关联的报价组成配置
        if component_ids:
            placeholders = ", ".join(["%s"] * len(component_ids))
            cur.execute(f"SELECT * FROM quote_component WHERE component_id IN ({placeholders})", component_ids)
        else:
            cur.execute("SELECT * FROM quote_component WHERE 1=0")
        components = cur.fetchall()

        # 2.5 查询选中的工艺配置
        if process_ids:
            placeholders = ", ".join(["%s"] * len(process_ids))
            cur.execute(f"SELECT * FROM process_config WHERE process_id IN ({placeholders})", process_ids)
        else:
            cur.execute("SELECT * FROM process_config WHERE 1=0")
        processes = cur.fetchall()

        # 2.6 查询选中的BOM物料（灵活引用，用于公式计算）
        bom_dict = {}  # 存储BOM物料（key=bom_price_+bom_id，value=物料单价）
        bom_details = []  # BOM物料明细（用于报价单展示）
        if final_bom_ids:
            placeholders = ", ".join(["%s"] * len(final_bom_ids))
            cur.execute(f"SELECT * FROM bom_material WHERE bom_id IN ({placeholders}) AND is_active = 1", final_bom_ids)
            boms = cur.fetchall()
            for bom in boms:
                bom_key = f"bom_price_{bom['bom_id']}"  # 公式中引用BOM的格式：bom_price_1
                bom_dict[bom_key] = bom["bom_price"]
                bom_details.append({
                    "bom_id": bom["bom_id"],
                    "bom_name": bom["bom_name"],
                    "bom_price": bom["bom_price"],
                    "price_unit": bom["price_unit"]
